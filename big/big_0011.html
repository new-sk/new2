oddcount ← function(x) {

k ← 0

for (n in x) if (n %% 2 ==1 ) k<-k+1

return(k)

}

oddcount2 ← function(x) {

k ← 0

for (i in 1:length(x)) if (x[i] %% 2 == 1) k ← k+1

return(k)

}

x ← c(1,2,3)

x ← integer() # integer type인데, 길이는 0

x ← "integer()"
