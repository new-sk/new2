## 1.2 실제 사항

<!DOCTYPE html>                   <!-- HTML5 문서 선언 -->

<html>                            <!-- html 시작, 종료 -->

<head>                            <!-- 공통선언 -->

<meta charset="UTF-8">            <!-- 한글선언. 깨짐방지 -->

<title>첫번째실습</title>    <!-- 탭 제목 -->

</head> 

<body>                            <!-- 눈에 보이는 영역 -->

<h1 style="color:navy">Hello World!</h1>

</body>

</html>
